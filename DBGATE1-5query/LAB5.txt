CREATE TABLE bus_status (
    bus_status_id VARCHAR(10) PRIMARY KEY,
    bus_number    VARCHAR(10),
    status_id     VARCHAR(10),
    FOREIGN KEY (bus_number) REFERENCES Bus(Bus_no),
    FOREIGN KEY (status_id)  REFERENCES status(status_ID)
);

SELECT 
    co.Owner_ID,
    co.Owner_name,
    co.Contacts,
    bc.Company_name,
    cbr.Bus_no,
    b.Source,
    b.Destination,
    b.Couch_type,
    b.Fair,
    s.status        AS Bus_Status,
    s.description   AS Status_Description
FROM company_owner co
INNER JOIN company_owner_relation cor ON co.Owner_ID    = cor.Owner_ID
INNER JOIN bus_company            bc  ON cor.Company_ID = bc.Company_ID
INNER JOIN company_bus_relation   cbr ON bc.Company_ID  = cbr.Company_ID
INNER JOIN Bus                    b   ON cbr.Bus_no     = b.Bus_no
INNER JOIN bus_status             bs  ON b.Bus_no       = bs.bus_number
INNER JOIN status                 s   ON bs.status_id   = s.status_ID
ORDER BY co.Owner_ID;

SELECT 
    bc.Company_ID,
    bc.Company_name,
    bc.Address,
    cbr.Bus_no,
    b.Source,
    b.Destination,
    b.Couch_type,
    b.Fair,
    b.SeatCapacity,
    s.status        AS Bus_Status,
    s.description   AS Status_Description
FROM bus_company bc
INNER JOIN company_bus_relation cbr ON bc.Company_ID = cbr.Company_ID
INNER JOIN Bus                  b   ON cbr.Bus_no    = b.Bus_no
INNER JOIN bus_status           bs  ON b.Bus_no      = bs.bus_number
INNER JOIN status               s   ON bs.status_id  = s.status_ID
ORDER BY bc.Company_ID;
